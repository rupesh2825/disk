pragma solidity ^0.8.0;

contract StudentRegistry {
    struct Student {
        string name;
        string course;
        bool isRegistered;
    }

    mapping(uint => Student) private students;

    function registerStudent(uint _rollNo, string memory _name, string memory _course) public {
        require(!students[_rollNo].isRegistered, "Student already registered");

        students[_rollNo] = Student({
            name: _name,
            course: _course,
            isRegistered: true
        });
    }

    function getStudent(uint _rollNo) public view returns (string memory, string memory) {
        require(students[_rollNo].isRegistered, "Student not registered");
        Student memory s = students[_rollNo];
        return (s.name, s.course);
    }

    function isStudentRegistered(uint _rollNo) public view returns (bool) {
        return students[_rollNo].isRegistered;
    }
}
